#include <iostream>
using namespace std;

// Function to print N Fibonacci numbers (Iterative)
void fibonacciNonRecursive(int N) {
    if (N <= 0) {
        cout << "Please enter a positive integer." << endl;
        return;
    }

    int num1 = 0, num2 = 1;
    cout << "Fibonacci Series (Non-Recursive): ";
    for (int i = 0; i < N; i++) {
        cout << num1 << " ";
        int next = num1 + num2;
        num1 = num2;
        num2 = next;
    }
    cout << endl;
}

int main() {
    int N;
    cout << "Enter the number of terms: ";
    cin >> N;
    fibonacciNonRecursive(N);
    return 0;
}
